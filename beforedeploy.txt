CREATE TABLE rework_pass (
    id INT AUTO_INCREMENT PRIMARY KEY,
    chassis_sn VARCHAR(100)
      CHARACTER SET utf8mb4
      COLLATE utf8mb4_general_ci
      NOT NULL,
    notes TEXT,
    CONSTRAINT fk_rework_pass_chassis
    FOREIGN KEY (chassis_sn)
    REFERENCES builds(chassis_sn)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

  chassis_sn VARCHAR(100)
      CHARACTER SET utf8mb4

      NOT NULL,
  pass_fail ENUM('Pass', 'Fail'),
  notes TEXT,
  dimm VARCHAR(10),
  bmc VARCHAR(10),
  m2 VARCHAR(10),
  liquid_cooler VARCHAR(10),
  location ENUM('Metcenter', 'Ceva', 'GUI Sheet location', 'Oracle'),

  status ENUM('Available', 'Delivered'),
  CONSTRAINT fk_rma_chassis
    FOREIGN KEY (chassis_sn)
    REFERENCES builds(chassis_sn)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);
